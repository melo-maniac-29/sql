declare
d dep.id%type;
nm empl.ename%type;
i empl.eid%type;
s empl.salary%type;
t empl.salary%type;
u empl.salary%type;
cursor cseresearch is select ename,eid,salary from empl,dep where dep_name='cs' and months_between(sysdate,doj)/12>10 and dep.id=empl.depid;
begin
open cseresearch;
loop
fetch cseresearch into nm,i,s;
exit when cseresearch%notfound;
t:=s;
s:=s+(s*0.05);
u:=s-t;
update empl set salary=s where eid=i;
insert into updated_table values(nm,u);
end loop;
close cseresearch;
end;
